CREATE DATABASE pawcare

CREATE TABLE user_credentials (
	id INT AUTO_INCREMENT PRIMARY KEY,
    userName varchar(50) NOT NULL UNIQUE,
    email VARCHAR(50) NOT NULL UNIQUE,
    userRole ENUM('User', 'Veterinarian', 'Admin') NOT NULL DEFAULT 'User',
    password VARCHAR(225) NOT NULL,
    created_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP
)

CREATE TABLE user_infos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_ID INT NOT NULL,
	firstName VARCHAR(50) NULL,
    middleName VARCHAR(50) NULL,
    lastName VARCHAR(50) NULL,
    suffix VARCHAR(15) NULL,
    phoneNumber VARCHAR(20) NULL,
    houseNum VARCHAR(50) NULL,
    province VARCHAR(50) NULL,
    municipality VARCHAR(50) NULL,
    barangay VARCHAR(50) NULL,
    zipCode VARCHAR(10) NULL,
    
    CONSTRAINT fk_user
    	FOREIGN KEY (user_ID) REFERENCES user_credentials(id)
    	ON DELETE CASCADE
    	ON UPDATE CASCADE
)
